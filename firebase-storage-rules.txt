/* Paste this into Firebase Console → Storage → Rules (or deploy with Firebase CLI) */

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow public read only for branding/dashboard_buttons
    match /branding/dashboard_buttons/{allPaths=**} {
      allow read: if true;
    }
    // Everything else requires auth
    match /{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}
